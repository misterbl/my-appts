export interface IState {books: [] | null};
 
 
export const saveBooks = (
  state: IState['books'] = null,
  action: AnyAction,
): IState['books'] => { 
  switch (action.type) {
    case 'SAVE_BOOKS':
      return action.data;
    default:
      return state;
  }
};
